interface Car {
    virtual drive();
}

class Tesla : Car {
    drive() {
        useBattery();
    }
}

class Porsche : Car {
    drive() {
        useBenzin();
    }
}

Car cars[] = {Tesla(), Porsche()};

for (car in cars) {
    car.drive();
}
